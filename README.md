encrypt text with one key

INSTALL
```
npm install rotten-encryption2 -S
```

 ### USAGE
 ```
var rotten = require('../index')('mars')
var code = rotten.encode('mars森') // '_>5%1tm1Tp1Tt'
rotten.decode(code) // 'mars森'
 ```
